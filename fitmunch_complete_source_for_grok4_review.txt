// Modern App Initialization
class FitMunchApp {
  constructor() {
    this.state = window.fitmunchState;

  }

  async init() {
    console.log('ðŸŽ¯ Initializing Enhanced FitMunch App...');
    
    try {

      this.showEmergencyFallback();
    }
  }

  showMainApp() {
    const fallback = document.getElementById('emergencyFallback');
    const mainApp = document.getElementById('mainApp');
    
    if (fallback) fallback.style.display = 'none';
    if (mainApp) {
      mainApp.style.display = 'block';
      mainApp.classList.add('animate__animated', 'animate__fadeIn');
    }
  }

  showEmergencyFallback() {
    const fallback = document.getElementById('emergencyFallback');
    const mainApp = document.getElementById('mainApp');
    
    if (fallback) fallback.style.display = 'flex';
    if (mainApp) mainApp.style.display = 'none';
  }

  async setupTheme() {
    const themeToggle = document.getElementById('themeToggle');
    const savedTheme = localStorage.getItem('fitmunch_theme') || 'light';
    
    document.documentElement.setAttribute('data-bs-theme', savedTheme);
    if (themeToggle) {
      themeToggle.checked = savedTheme === 'dark';
      themeToggle.addEventListener('change', this.toggleTheme.bind(this));
    }
  }

  toggleTheme() {
    const currentTheme = document.documentElement.getAttribute('data-bs-theme');
    const newTheme = currentTheme === 'dark' ? 'light' : 'dark';
    
    document.documentElement.setAttribute('data-bs-theme', newTheme);
    localStorage.setItem('fitmunch_theme', newTheme);
    
    // Reinitialize charts with new theme
    setTimeout(() => this.updateChartsTheme(), 100);
  }

  async initializeCharts() {
    // Macro Chart
    await this.initMacroChart();
    
    // Wellness Score Animation
    this.animateWellnessScore();
  }

  async initMacroChart() {
    const canvas = document.getElementById('macroChart');
    if (!canvas) return;

    const ctx = canvas.getContext('2d');
    const isDark = document.documentElement.getAttribute('data-bs-theme') === 'dark';
    
    this.charts.macro = new Chart(ctx, {
      type: 'doughnut',
      data: {
        labels: ['Carbs', 'Protein', 'Fat'],
        datasets: [{
          data: [this.state.dailyLog.macros.carbs, this.state.dailyLog.macros.protein, this.state.dailyLog.macros.fat],
          backgroundColor: [
            '#3b82f6', // Blue for carbs
            '#10b981', // Green for protein  
            '#f59e0b'  // Orange for fat
          ],
          borderWidth: 3,
          borderColor: isDark ? '#1f2937' : '#ffffff',
          hoverBorderWidth: 4,
          cutout: '70%'
        }]
      },
      options: {
        responsive: true,
        maintainAspectRatio: false,
        plugins: {
          legend: {
            display: false
          },
          tooltip: {
            backgroundColor: isDark ? '#1f2937' : '#ffffff',
            titleColor: isDark ? '#f9fafb' : '#111827',
            bodyColor: isDark ? '#f9fafb' : '#111827',
            borderColor: isDark ? '#374151' : '#e5e7eb',
            borderWidth: 1,
            cornerRadius: 8,
            displayColors: true,
            callbacks: {
              label: function(context) {
                const label = context.label || '';
                const value = context.parsed;
                const total = context.dataset.data.reduce((a, b) => a + b, 0);
                const percentage = Math.round((value / total) * 100);
                return `${label}: ${value}g (${percentage}%)`;
              }
            }
          }
        },
        animation: {
          animateRotate: true,
          animateScale: true,
          duration: 2000,
          easing: 'easeOutQuart'
        }
      }
    });
  }

  animateWellnessScore() {
    const circle = document.getElementById('wellnessCircle');
    const scoreElement = document.getElementById('wellnessScore');
    
    if (!circle || !scoreElement) return;

    const score = this.calculateWellnessScore();
    const circumference = 2 * Math.PI * 65; // radius = 65
    const offset = circumference - (score / 100) * circumference;
    
    // Animate the circle
    circle.style.strokeDasharray = circumference;
    circle.style.strokeDashoffset = circumference;
    
    setTimeout(() => {
      circle.style.strokeDashoffset = offset;
    }, 500);

    // Animate the score number
    this.animateNumber(scoreElement, 0, score, 2000);
  }

  animateNumber(element, start, end, duration) {
    const range = end - start;
    const increment = range / (duration / 16);
    let current = start;
    
    const timer = setInterval(() => {
      current += increment;
      if ((increment > 0 && current >= end) || (increment < 0 && current <= end)) {
        current = end;
        clearInterval(timer);
      }
      element.textContent = Math.round(current);
    }, 16);
  }

  calculateWellnessScore() {
    const { calories, steps } = this.state.dailyLog;
    const { calorieGoal, stepGoal } = this.state.user;
    
    let score = 50; // Base score
    
    // Calorie adherence (0-35 points)
    const calorieRatio = calories / calorieGoal;
    if (calorieRatio >= 0.8 && calorieRatio <= 1.1) {
      score += 35;
    } else if (calorieRatio >= 0.6 && calorieRatio <= 1.3) {
      score += 25;
    } else {
      score += 10;
    }
    
    // Step goal (0-25 points)
    const stepRatio = steps / stepGoal;
    if (stepRatio >= 1.0) {
      score += 25;
    } else if (stepRatio >= 0.7) {
      score += 15;
    } else {
      score += 5;
    }
    
    // Water intake (0-15 points)
    if (this.state.dailyLog.water >= 8) {
      score += 15;
    } else if (this.state.dailyLog.water >= 6) {
      score += 10;
    } else {
      score += 5;
    }
    
    return Math.min(Math.max(score, 0), 100);
  }

  async setupNavigation() {
    // Desktop navigation
    const navButtons = document.querySelectorAll('.nav-pills .nav-link[data-section]');
    navButtons.forEach(button => {
      button.addEventListener('click', (e) => {
        e.preventDefault();
        const section = button.getAttribute('data-section');
        this.showSection(section);
        this.updateNavState(button);
      });
    });

    // Mobile navigation
    const mobileNavButtons = document.querySelectorAll('.nav-mobile-btn[data-section]');
    mobileNavButtons.forEach(button => {
      button.addEventListener('click', (e) => {
        e.preventDefault();
        const section = button.getAttribute('data-section');
        this.showSection(section);
        this.updateMobileNavState(button);
      });
    });

    // Show dashboard by default
    this.showSection('dashboard');
  }

  showSection(sectionId) {
    console.log(`ðŸ“ Navigating to: ${sectionId}`);
    
    // Hide all sections
    const sections = document.querySelectorAll('.app-section');
    sections.forEach(section => {
      section.classList.remove('active');
    });
    
    // Show target section
    const targetSection = document.getElementById(sectionId);
    if (targetSection) {
      targetSection.classList.add('active');
      
      // Initialize section-specific features
      this.initializeSection(sectionId);
      
      // Smooth scroll to top
      window.scrollTo({ top: 0, behavior: 'smooth' });
    }
  }

  updateNavState(activeButton) {
    const navButtons = document.querySelectorAll('.nav-pills .nav-link');
    navButtons.forEach(btn => btn.classList.remove('active'));
    activeButton.classList.add('active');
  }

  updateMobileNavState(activeButton) {
    const mobileButtons = document.querySelectorAll('.nav-mobile-btn');
    mobileButtons.forEach(btn => btn.classList.remove('active'));
    activeButton.classList.add('active');
  }

  async initializeSection(sectionId) {
    switch(sectionId) {
      case 'dashboard':
        this.updateDashboard();
        break;
      case 'meal':
        this.initializeMealSection();
        break;
      case 'workout':
        this.initializeWorkoutSection();
        break;
      case 'food':
        this.initializeFoodSection();
        break;
      case 'shopping':
        this.initializeShoppingSection();
        break;
    }
  }

  updateDashboard() {
    // Update progress bars and stats
    this.updateProgressBars();
    this.updateStats();
  }

  updateProgressBars() {
    const { calories, steps } = this.state.dailyLog;
    const { calorieGoal, stepGoal } = this.state.user;
    
    // Find progress bars and update them
    const calorieProgress = (calories / calorieGoal) * 100;
    const stepProgress = (steps / stepGoal) * 100;
    
    // Update with animation
    this.animateProgressBar('.stat-mini-card:nth-child(1) .progress-bar', calorieProgress);
    this.animateProgressBar('.stat-mini-card:nth-child(2) .progress-bar', stepProgress);
  }

  animateProgressBar(selector, percentage) {
    const progressBar = document.querySelector(selector);
    if (!progressBar) return;
    
    progressBar.style.width = '0%';
    setTimeout(() => {
      progressBar.style.width = `${Math.min(percentage, 100)}%`;
    }, 100);
  }

  updateStats() {
    // Update displayed stats with current data
    const elements = {
      calorieDisplay: `${this.state.dailyLog.calories} / ${this.state.user.calorieGoal}`,
      stepsDisplay: `${this.state.dailyLog.steps} / ${this.state.user.stepGoal}`,
    };
    
    Object.entries(elements).forEach(([id, value]) => {
      const element = document.getElementById(id);
      if (element) element.textContent = value;
    });
  }

  async setupAnimations() {
    // Intersection Observer for scroll animations
    const observerOptions = {
      threshold: 0.1,
      rootMargin: '0px 0px -50px 0px'
    };

    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          entry.target.classList.add('animate__animated', 'animate__fadeInUp');
        }
      });
    }, observerOptions);

    // Observe bento items
    const bentoItems = document.querySelectorAll('.bento-item');
    bentoItems.forEach(item => observer.observe(item));
  }

  async loadUserData() {
    // Load and apply user preferences
    this.personalizeUI();
    this.updateProfileDisplay();
  }

  personalizeUI() {
    const { goal } = this.state.user;
    
    // Adjust UI based on user goal
    switch(goal) {
      case 'weight_loss':
        document.documentElement.style.setProperty('--accent', '#ef4444'); // Red for intensity
        break;
      case 'muscle_gain':
        document.documentElement.style.setProperty('--accent', '#8b5cf6'); // Purple for strength
        break;
      case 'maintenance':
        document.documentElement.style.setProperty('--accent', '#10b981'); // Green for balance
        break;
    }
  }

  updateProfileDisplay() {
    const nameElement = document.querySelector('.hero-item h1');
    if (nameElement && this.state.user.name !== 'FitMunch User') {
      nameElement.textContent = `Welcome Back, ${this.state.user.name.split(' ')[0]}!`;
    }
  }

  setupEventListeners() {
    // Global error handling
    window.addEventListener('error', this.handleError.bind(this));
    window.addEventListener('unhandledrejection', this.handleError.bind(this));
    
    // Performance monitoring
    window.addEventListener('load', () => {
      const loadTime = performance.now();
      console.log(`âš¡ App loaded in ${loadTime.toFixed(2)}ms`);
    });
  }

  handleError(error) {
    console.error('ðŸš¨ App Error:', error);
    // Could show user-friendly error notification here
  }

  updateChartsTheme() {
    // Reinitialize charts with new theme
    if (this.charts.macro) {
      this.charts.macro.destroy();
      this.initMacroChart();
    }
  }

  // Save state to localStorage
  saveState() {
    localStorage.setItem('fitmunch_user', JSON.stringify(this.state.user));
    localStorage.setItem('fitmunch_daily_log', JSON.stringify(this.state.dailyLog));
  }
}

function showNotification(message, type = 'info') {
  const notification = document.createElement('div');
  notification.className = `notification position-fixed bottom-0 end-0 m-4 alert alert-${type === 'success' ? 'success' : 'info'} animate__animated animate__slideInRight`;
  notification.innerHTML = `
    <div class="d-flex align-items-center gap-2">
      <i class="fas fa-${type === 'success' ? 'check-circle' : 'info-circle'}"></i>
      <span>${message}</span>
    </div>
  `;
  
  document.body.appendChild(notification);
  
  setTimeout(() => {
    notification.classList.add('animate__slideOutRight');
    setTimeout(() => notification.remove(), 300);
  }, 3000);
}
